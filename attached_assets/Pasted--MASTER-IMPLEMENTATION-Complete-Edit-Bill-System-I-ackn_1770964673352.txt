import { db, auth } from '../lib/firebase';
import { 
  doc, 
  updateDoc, 
  serverTimestamp, 
  collection,
  addDoc,
  getDoc,
  query,
  where,
  getDocs
} from 'firebase/firestore';

/**
 * Update a bill with validation and ownership check
 */
export async function updateBill(billId, data, previousData) {
  const user = auth.currentUser;
  if (!user) throw new Error('Not authenticated');

  const billRef = doc(db, 'bills', billId);
  const billSnap = await getDoc(billRef);
  
  if (!billSnap.exists()) throw new Error('Bill not found');
  if (billSnap.data().userId !== user.uid) throw new Error('Unauthorized');

  // Strip any forbidden fields
  const { userId, createdAt, ...safeData } = data;
  
  const updatedFields = {
    ...safeData,
    updatedAt: serverTimestamp()
  };

  await updateDoc(billRef, updatedFields);
  
  // Log activity
  await logBillUpdate(user.uid, {
    billId,
    billerName: data.billerName || previousData.billerName,
    previousAmount: previousData.amount,
    newAmount: data.amount,
    previousDueDate: previousData.dueDate,
    newDueDate: data.dueDate
  });

  // Trigger forecast recalculation
  await recalculateForecast(user.uid);

  return { success: true, billId };
}

/**
 * Log bill update activity
 */
export async function logBillUpdate(userId, details) {
  const activityRef = collection(db, 'users', userId, 'activity');
  
  await addDoc(activityRef, {
    type: 'bill_updated',
    billId: details.billId,
    billerName: details.billerName,
    previousAmount: details.previousAmount,
    newAmount: details.newAmount,
    previousDueDate: details.previousDueDate,
    newDueDate: details.newDueDate,
    timestamp: serverTimestamp()
  });
}

/**
 * Recalculate 30/60/90 day forecast
 * Mock implementation - expand based on your forecasting logic
 */
export async function recalculateForecast(userId) {
  const billsRef = collection(db, 'bills');
  const q = query(billsRef, where('userId', '==', userId));
  const snapshot = await getDocs(q);
  
  const upcomingBills = snapshot.docs.map(doc => ({
    id: doc.id,
    ...doc.data()
  }));

  // Simple 30-day forecast
  const now = new Date();
  const thirtyDays = now.setDate(now.getDate() + 30);
  
  const forecast30Day = upcomingBills
    .filter(bill => bill.dueDate?.toDate() <= thirtyDays)
    .reduce((sum, bill) => sum + (bill.amount || 0), 0);

  // Store forecast in user profile or separate collection
  const userRef = doc(db, 'users', userId);
  await updateDoc(userRef, {
    forecast30Day,
    forecastLastCalculated: serverTimestamp()
  });

  return forecast30Day;
}

/**
 * Validate bill data before update
 */
export function validateBillData(data) {
  const errors = {};
  
  if (!data.amount || data.amount <= 0) {
    errors.amount = 'Amount must be greater than 0';
  }
  
  if (!data.dueDate) {
    errors.dueDate = 'Due date is required';
  }
  
  if (data.billerName?.trim() === '') {
    errors.billerName = 'Biller name is required';
  }

  return {
    isValid: Object.keys(errors).length === 0,
    errors
  };
}

/**
 * Advanced forecast recalculation with multiple time horizons
 */
export async function recalculateForecast(userId) {
  if (!userId) throw new Error('User ID required');

  const billsRef = collection(db, 'bills');
  const q = query(billsRef, where('userId', '==', userId));
  const snapshot = await getDocs(q);
  
  const now = new Date();
  const bills = snapshot.docs.map(doc => ({
    id: doc.id,
    ...doc.data()
  }));

  // Calculate 30/60/90 day forecasts
  const forecast30 = calculateHorizon(bills, now, 30);
  const forecast60 = calculateHorizon(bills, now, 60);
  const forecast90 = calculateHorizon(bills, now, 90);
  
  // Calculate cash flow stress indicators
  const recurringTotal = bills
    .filter(b => b.isRecurring)
    .reduce((sum, b) => sum + (b.amount || 0), 0);
  
  const upcomingCount = bills.filter(b => {
    const due = b.dueDate?.toDate?.() || new Date(b.dueDate);
    return due >= now && due <= addDays(now, 30);
  }).length;

  // Store forecast in user document
  const userRef = doc(db, 'users', userId);
  await updateDoc(userRef, {
    forecast30,
    forecast60,
    forecast90,
    recurringMonthlyTotal: recurringTotal,
    upcomingBillsCount: upcomingCount,
    forecastLastCalculated: serverTimestamp(),
    forecastVersion: '2.0.0'
  });

  return {
    forecast30,
    forecast60,
    forecast90,
    recurringTotal,
    upcomingCount
  };
}

function calculateHorizon(bills, startDate, days) {
  const horizon = addDays(startDate, days);
  
  return bills
    .filter(bill => {
      const dueDate = bill.dueDate?.toDate?.() || new Date(bill.dueDate);
      return dueDate >= startDate && dueDate <= horizon;
    })
    .reduce((sum, bill) => sum + (bill.amount || 0), 0);
}

function addDays(date, days) {
  const result = new Date(date);
  result.setDate(result.getDate() + days);
  return result;

import { collection, addDoc, serverTimestamp, query, where, orderBy, limit, getDocs } from 'firebase/firestore';
import { db } from '../lib/firebase';

export const ActivityTypes = {
  BILL_CREATED: 'bill_created',
  BILL_UPDATED: 'bill_updated',
  BILL_DELETED: 'bill_deleted',
  BILL_PAID: 'bill_paid',
  FORECAST_RUN: 'forecast_run',
  PAYMENT_REDIRECT: 'payment_redirect'
};

/**
 * Log any user activity
 */
export async function logActivity(userId, activity) {
  if (!userId) throw new Error('User ID required');
  
  const activityRef = collection(db, 'users', userId, 'activity');
  
  const activityDoc = {
    type: activity.type,
    timestamp: serverTimestamp(),
    ...activity.data,
    userAgent: typeof window !== 'undefined' ? window.navigator.userAgent : null,
    platform: typeof window !== 'undefined' ? window.navigator.platform : null
  };

  return await addDoc(activityRef, activityDoc);
}

/**
 * Get recent activity for a user
 */
export async function getRecentActivity(userId, maxItems = 20) {
  const activityRef = collection(db, 'users', userId, 'activity');
  const q = query(
    activityRef,
    orderBy('timestamp', 'desc'),
    limit(maxItems)
  );
  
  const snapshot = await getDocs(q);
  return snapshot.docs.map(doc => ({
    id: doc.id,
    ...doc.data()
  }));
}

/**
 * Log bill payment redirect
 */
export async function logPaymentRedirect(userId, billId, billerName) {
  return logActivity(userId, {
    type: ActivityTypes.PAYMENT_REDIRECT,
    data: {
      billId,
      billerName,
      timestamp: serverTimestamp()
    }
  });
}

/**
 * Log bill marked as paid
 */
export async function logBillPaid(userId, billId, amount) {
  return logActivity(userId, {
    type: ActivityTypes.BILL_PAID,
    data: {
      billId,
      amount,
      timestamp: serverTimestamp()
    }
  });
}
