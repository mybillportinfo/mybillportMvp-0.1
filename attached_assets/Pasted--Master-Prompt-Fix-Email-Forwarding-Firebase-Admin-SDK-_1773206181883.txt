✅ Step 1: Update Firebase Admin SDK Initialization

```typescript
import admin from 'firebase-admin';

// Initialize Firebase Admin SDK safely
function initializeFirebaseAdmin() {
  // Check if already initialized
  if (admin.apps.length > 0) {
    return admin;
  }

  try {
    // Try environment variable first (Vercel)
    const serviceAccountStr = process.env.FIREBASE_SERVICE_ACCOUNT_KEY || process.env.FIREBASE_SA_JSON;
    
    if (serviceAccountStr && serviceAccountStr.trim().startsWith('{')) {
      // Parse the JSON string
      const serviceAccount = JSON.parse(serviceAccountStr);
      
      admin.initializeApp({
        credential: admin.credential.cert(serviceAccount),
        databaseURL: `https://${serviceAccount.project_id}.firebaseio.com`
      });
      
      console.log('✅ Firebase Admin initialized with service account JSON');
      return admin;
    }

    // Fallback: use default credentials (for local development)
    console.log('⚠️ Using default Firebase Admin credentials (local only)');
    admin.initializeApp({
      credential: admin.credential.applicationDefault()
    });
    
    return admin;
  } catch (error) {
    console.error('❌ Failed to initialize Firebase Admin:', error);
    throw new Error('Firebase Admin initialization failed');
  }
}

const adminApp = initializeFirebaseAdmin();
export const adminDb = adminApp.firestore();
export const adminAuth = adminApp.auth();
export default adminApp;
```

✅ Step 2: Fix Email Forwarding Webhook

pages/api/email-forward.ts

```typescript
import { NextApiRequest, NextApiResponse } from 'next';
import { adminDb } from '../../../lib/firebaseAdmin';
import { extractBillData } from '../../../lib/extractBillData';

export const config = {
  api: {
    bodyParser: true,
    sizeLimit: '10mb',
  },
};

export default async function handler(req: NextApiRequest, res: NextApiResponse) {
  // Log everything for debugging
  console.log('📧 Email webhook received:', {
    method: req.method,
    timestamp: new Date().toISOString(),
    headers: req.headers['content-type'],
  });

  // Only accept POST requests
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Method not allowed' });
  }

  try {
    // Parse SendGrid inbound format
    const { to, from, subject, text, html, attachments } = req.body;

    // Extract the email alias (handle array or string)
    const emailAlias = Array.isArray(to) ? to[0] : to;
    
    if (!emailAlias) {
      console.error('❌ No recipient email found');
      return res.status(400).json({ error: 'No recipient email' });
    }

    console.log('📨 Processing email:', {
      to: emailAlias,
      from,
      subject,
      textLength: text?.length || 0,
      hasAttachments: attachments?.length > 0
    });

    // Look up user by email alias
    const aliasSnapshot = await adminDb.collection('emailAliases')
      .doc(emailAlias)
      .get();

    if (!aliasSnapshot.exists) {
      console.log('❌ No user found for alias:', emailAlias);
      // Still return 200 to prevent SendGrid from retrying
      return res.status(200).json({ message: 'Alias not found - ignored' });
    }

    const userId = aliasSnapshot.data()?.userId;
    console.log('✅ Found user:', userId);

    // Extract bill data using your existing extraction logic
    console.log('🤖 Extracting bill data...');
    const extractedData = await extractBillData({
      from,
      subject,
      text,
      html,
      attachments
    });

    console.log('📊 Extracted data:', extractedData);

    // Store in pending bills (using Admin SDK to bypass rules)
    const pendingRef = adminDb
      .collection('users')
      .doc(userId)
      .collection('pendingBills')
      .doc();

    await pendingRef.set({
      ...extractedData,
      sourceEmail: from,
      sourceSubject: subject,
      receivedAt: admin.firestore.FieldValue.serverTimestamp(),
      status: 'pending',
      createdAt: admin.firestore.FieldValue.serverTimestamp()
    });

    console.log('✅ Pending bill saved:', pendingRef.id);

    return res.status(200).json({ 
      success: true,
      message: 'Email processed successfully',
      pendingId: pendingRef.id 
    });

  } catch (error) {
    console.error('❌ Error processing email:', error);
    // Still return 200 to prevent SendGrid from retrying indefinitely
    return res.status(200).json({ 
      error: 'Failed to process email',
      message: error.message 
    });
  }
}
```

✅ Step 4: Fix Firestore Security Rules

```javascript
// firestore.rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users can read/write their own data
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Pending bills subcollection
    match /users/{userId}/pendingBills/{billId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Email aliases - readable by anyone (for webhook), but only Admin SDK writes
    match /emailAliases/{alias} {
      allow read: if true;  // Webhook needs to read this
      allow write: if false; // Only Admin SDK via backend
    }
    
    // Referral codes
    match /referralCodes/{code} {
      allow read: if request.auth != null;
      allow write: if false; // Only Admin SDK
    }
  }
}
```
