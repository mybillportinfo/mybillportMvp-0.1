📌 Problem Diagnosis

Issue Root Cause
Referral code spins forever createReferralCode writes to Firestore via client SDK – security rules block silently, spinner never stops.

🛠️ Implementation Tasks

1. Create Server‑Side Referral Code API

File: pages/api/referral/create-code.ts

```typescript
import { NextApiRequest, NextApiResponse } from 'next';
import { adminDb } from '../../../lib/firebaseAdmin';
import { verifyFirebaseToken } from '../../../lib/auth';

export default async function handler(req: NextApiRequest, res: NextApiResponse) {
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Method not allowed' });
  }

  try {
    // Verify Firebase token
    const token = req.headers.authorization?.split('Bearer ')[1];
    if (!token) {
      return res.status(401).json({ error: 'Unauthorized' });
    }

    const decoded = await verifyFirebaseToken(token);
    const uid = decoded.uid;

    // Check if user already has a referral code
    const userRef = adminDb.collection('userProfiles').doc(uid);
    const userDoc = await userRef.get();

    if (userDoc.exists && userDoc.data()?.referralCode) {
      return res.status(200).json({ 
        referralCode: userDoc.data()?.referralCode 
      });
    }

    // Generate code from username or display name
    const username = req.body.username || userDoc.data()?.username || uid.slice(0, 6);
    const baseCode = username.replace(/[^a-zA-Z0-9]/g, '').toUpperCase().slice(0, 8);
    const uniqueCode = `${baseCode}-${Math.random().toString(36).substring(2, 6).toUpperCase()}`;

    // Store in Firestore (batch write for consistency)
    const batch = adminDb.batch();

    // Create referral code document
    const codeRef = adminDb.collection('referralCodes').doc(uniqueCode);
    batch.set(codeRef, {
      code: uniqueCode,
      ownerId: uid,
      createdAt: new Date(),
      timesUsed: 0,
      active: true
    });

    // Update user profile
    batch.set(userRef, { 
      referralCode: uniqueCode,
      updatedAt: new Date()
    }, { merge: true });

    await batch.commit();

    return res.status(200).json({ referralCode: uniqueCode });

  } catch (error) {
    console.error('Error creating referral code:', error);
    return res.status(500).json({ error: 'Failed to create referral code' });
  }
}
```

2. Create Server‑Side Email Alias API

File: pages/api/user/create-email-alias.ts

```typescript
import { NextApiRequest, NextApiResponse } from 'next';
import { adminDb } from '../../../lib/firebaseAdmin';
import { verifyFirebaseToken } from '../../../lib/auth';

export default async function handler(req: NextApiRequest, res: NextApiResponse) {
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Method not allowed' });
  }

  try {
    const token = req.headers.authorization?.split('Bearer ')[1];
    if (!token) {
      return res.status(401).json({ error: 'Unauthorized' });
    }

    const decoded = await verifyFirebaseToken(token);
    const uid = decoded.uid;

    // Check if alias already exists
    const userRef = adminDb.collection('userProfiles').doc(uid);
    const userDoc = await userRef.get();

    if (userDoc.exists && userDoc.data()?.emailAlias) {
      return res.status(200).json({ 
        emailAlias: userDoc.data()?.emailAlias 
      });
    }

    // Generate unique alias
    const shortId = uid.slice(0, 6);
    const timestamp = Date.now().toString(36).slice(-4);
    const alias = `bills.${shortId}.${timestamp}@mybillport.com`;

    // Store in Firestore
    await userRef.set({
      emailAlias: alias,
      emailAliasCreatedAt: new Date()
    }, { merge: true });

    // Also store in emailAliases collection for routing
    await adminDb.collection('emailAliases').doc(alias).set({
      userId: uid,
      createdAt: new Date(),
      active: true
    });

    return res.status(200).json({ emailAlias: alias });

  } catch (error) {
    console.error('Error creating email alias:', error);
    return res.status(500).json({ error: 'Failed to create email alias' });
  }
}
```

3. Update DataContext to Use Server APIs

File: lib/DataContext.tsx

Find and replace the client‑side calls:

const createReferralCode = async () => {

After (fixed):

```typescript
const fetchReferralCode = async () => {
  if (!user) return;
  
  try {
    setLoading(true);
    const token = await user.getIdToken();
    const response = await fetch('/api/referral/create-code', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${token}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({ username: user.displayName })
    });
    
    const data = await response.json();
    if (response.ok) {
      setReferralCode(data.referralCode);
    } else {
      console.error('Failed to get referral code:', data.error);
    }
  } catch (error) {
    console.error('Error fetching referral code:', error);
  } finally {
    setLoading(false);
  }
};

const fetchEmailAlias = async () => {
  if (!user) return;
  
  try {
    const token = await user.getIdToken();
    const response = await fetch('/api/user/create-email-alias', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${token}`
      }
    });
    
    const data = await response.json();
    if (response.ok) {
      setEmailAlias(data.emailAlias);
    }
  } catch (error) {
    console.error('Error fetching email alias:', error);
  }
};
```
