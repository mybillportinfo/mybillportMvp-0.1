🔍 Step 2: Check Webhook Logs

Add detailed logging to your webhook endpoint:

```typescript
// pages/api/email-forward.ts
import { NextApiRequest, NextApiResponse } from 'next';
import { adminDb } from '../../../lib/firebaseAdmin';
import { extractBillData } from '../../../lib/extractBillData';

export default async function handler(req: NextApiRequest, res: NextApiResponse) {
  console.log('📧 Email webhook received:', {
    method: req.method,
    headers: req.headers,
    bodyType: typeof req.body,
    timestamp: new Date().toISOString()
  });

  if (req.method !== 'POST') {
    console.log('❌ Invalid method:', req.method);
    return res.status(405).json({ error: 'Method not allowed' });
  }

  try {
    // Parse the incoming email (SendGrid format)
    const { to, from, subject, text, html, attachments } = req.body;
    
    console.log('📨 Email details:', {
      to,
      from,
      subject,
      textLength: text?.length,
      hasAttachments: attachments?.length > 0
    });

    // Extract the email alias (to address)
    const emailAlias = Array.isArray(to) ? to[0] : to;
    
    // Look up user by email alias
    console.log('🔍 Looking up user for alias:', emailAlias);
    const aliasSnapshot = await adminDb.collection('emailAliases')
      .doc(emailAlias)
      .get();

    if (!aliasSnapshot.exists) {
      console.log('❌ No user found for alias:', emailAlias);
      return res.status(200).json({ message: 'Alias not found - ignoring' });
    }

    const userId = aliasSnapshot.data()?.userId;
    console.log('✅ Found user:', userId);

    // Extract bill data using your existing extraction logic
    console.log('🤖 Extracting bill data...');
    const extractedData = await extractBillData({
      from,
      subject,
      text,
      html,
      attachments
    });

    console.log('📊 Extracted data:', extractedData);

    // Store in pending bills
    const pendingRef = adminDb.collection('users').doc(userId)
      .collection('pendingBills').doc();

    await pendingRef.set({
      ...extractedData,
      sourceEmail: from,
      sourceSubject: subject,
      receivedAt: new Date(),
      status: 'pending',
      createdAt: new Date()
    });

    console.log('✅ Pending bill saved:', pendingRef.id);

    return res.status(200).json({ 
      message: 'Email processed successfully',
      pendingId: pendingRef.id 
    });

  } catch (error) {
    console.error('❌ Error processing email:', error);
    return res.status(500).json({ 
      error: 'Failed to process email',
      details: error.message 
    });
  }
}
```

🔍 Step 5: Update Frontend to Fetch Pending Bills

```typescript
// lib/firebase.ts
export const getPendingBills = async (userId: string) => {
  const snapshot = await db.collection('users').doc(userId)
    .collection('pendingBills')
    .where('status', '==', 'pending')
    .orderBy('receivedAt', 'desc')
    .get();

  return snapshot.docs.map(doc => ({
    id: doc.id,
    ...doc.data()
  }));
};
```

```typescript
// app/pending-bills/page.tsx
import { useEffect, useState } from 'react';
import { useAuth } from '../../hooks/useAuth';
import { getPendingBills } from '../../lib/firebase';

export default function PendingBillsPage() {
  const { user } = useAuth();
  const [bills, setBills] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    if (!user) return;

    const loadBills = async () => {
      try {
        setLoading(true);
        const pending = await getPendingBills(user.uid);
        setBills(pending);
      } catch (err) {
        console.error('Error loading pending bills:', err);
        setError('Could not load pending bills');
      } finally {
        setLoading(false);
      }
    };

    loadBills();
  }, [user]);

  if (loading) return <div>Loading pending bills...</div>;
  if (error) return <div className="text-red-500">{error}</div>;
  if (bills.length === 0) {
    return (
      <div className="text-center py-8">
        <p className="text-gray-500">No pending bills</p>
        <p className="text-sm text-gray-400">
          Bills detected from your forwarding email will appear here for review.
        </p>
        <a href="/settings" className="text-blue-500 hover:underline mt-4 inline-block">
          Set up email forwarding in Settings
        </a>
      </div>
    );
  }

  return (
    <div>
      <h1 className="text-2xl font-bold mb-6">Pending Bills</h1>
      <div className="space-y-4">
        {bills.map(bill => (
          <BillReviewCard key={bill.id} bill={bill} />
        ))}
      </div>
    </div>
  );
}
```
