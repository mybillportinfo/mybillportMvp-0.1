```javascript
const functions = require('firebase-functions');
const Anthropic = require('@anthropic-ai/sdk');
const vision = require('@google-cloud/vision');

// Initialize clients
const anthropic = new Anthropic({
  apiKey: functions.config().anthropic.key,
});

// Initialize Vision client with credentials
let visionClient;
try {
  const credentials = functions.config().google.application_credentials;
  if (credentials && typeof credentials === 'string') {
    visionClient = new vision.ImageAnnotatorClient({
      credentials: JSON.parse(credentials)
    });
  } else {
    visionClient = new vision.ImageAnnotatorClient();
  }
} catch (error) {
  console.error('Failed to initialize Vision client:', error);
}

exports.scanBill = functions
  .region('northamerica-northeast1')
  .https.onCall(async (data, context) => {
    if (!context.auth) {
      throw new functions.https.HttpsError('unauthenticated', 'You must be logged in.');
    }

    const { imageBase64 } = data;
    if (!imageBase64) {
      throw new functions.https.HttpsError('invalid-argument', 'Image data required.');
    }

    try {
      // Step 1: Google Vision OCR
      console.log('Running Google Vision OCR...');
      const imageBuffer = Buffer.from(imageBase64, 'base64');
      const [visionResult] = await visionClient.textDetection({
        image: { content: imageBuffer.toString('base64') }
      });
      
      const textAnnotations = visionResult.textAnnotations;
      let extractedText = '';
      if (textAnnotations && textAnnotations.length > 0) {
        extractedText = textAnnotations[0].description;
        console.log('OCR successful. Text length:', extractedText.length);
      } else {
        throw new functions.https.HttpsError(
          'failed-precondition',
          'No text could be read from this image. Please ensure the bill is clear and try again.'
        );
      }

      // Step 2: Claude structuring
      console.log('Sending text to Claude for structuring...');
      const claudeResponse = await anthropic.messages.create({
        model: 'claude-3-sonnet-20240229',
        max_tokens: 1024,
        temperature: 0.1,
        messages: [
          {
            role: 'user',
            content: `Extract bill information from this text and return ONLY this JSON format:

Text from bill:
${extractedText}

Return this exact JSON structure:
{
  "merchant_name": "string (the company name)",
  "total_amount": number (just the number, no currency symbol),
  "due_date": "YYYY-MM-DD (if found, otherwise null)",
  "account_number": "string or null",
  "confidence": "high|medium|low"
}

Only return valid JSON, no other text.`
          }
        ]
      });

      const billData = JSON.parse(claudeResponse.content[0].text);
      console.log('Bill scan successful:', billData);

      return {
        ...billData,
        _debug: {
          ocrLength: extractedText.length,
          ocrPreview: extractedText.substring(0, 200) + '...'
        }
      };

    } catch (error) {
      console.error('Scan error:', error);
      if (error.message.includes('OCR')) {
        throw new functions.https.HttpsError(
          'failed-precondition',
          'Could not read text from image. Please ensure the photo is clear and well-lit.'
        );
      } else if (error.message.includes('JSON')) {
        throw new functions.https.HttpsError(
          'internal',
          'Could not understand bill format. Please try a different image.'
        );
      } else {
        throw new functions.https.HttpsError(
          'internal',
          'Failed to process bill. Please try again.'
        );
      }
    }
  });
```
