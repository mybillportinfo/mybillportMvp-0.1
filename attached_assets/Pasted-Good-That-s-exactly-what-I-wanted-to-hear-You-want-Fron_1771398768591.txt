📍 Step 3 — Put This Inside functions/index.js

Replace everything with this:

const { onDocumentCreated } = require("firebase-functions/v2/firestore");
const { MailerSend, EmailParams, Sender, Recipient } = require("mailersend");
const { defineSecret } = require("firebase-functions/params");

const mailerSendApiKey = defineSecret("MAILERSEND_API_KEY");

exports.sendFeedbackEmail = onDocumentCreated(
  {
    document: "feedback/{feedbackId}",
    secrets: [mailerSendApiKey],
  },
  async (event) => {
    const feedback = event.data.data();

    const mailerSend = new MailerSend({
      apiKey: mailerSendApiKey.value(),
    });

    const sentFrom = new Sender("no-reply@mybillport.com", "MyBillPort");

    const recipients = [
      new Recipient("mybillportinfo@gmail.com", "Admin"),
    ];

    const emailParams = new EmailParams()
      .setFrom(sentFrom)
      .setTo(recipients)
      .setSubject(`New Feedback: ${feedback.category}`)
      .setHtml(`
        <h2>New Feedback Received</h2>
        <p><strong>From:</strong> ${feedback.userEmail}</p>
        <p><strong>Category:</strong> ${feedback.category}</p>
        <p><strong>Message:</strong></p>
        <p>${feedback.message}</p>
      `);

    try {
      await mailerSend.email.send(emailParams);
      console.log("Email sent successfully.");
    } catch (error) {
      console.error("MailerSend error:", error);
    }
  }
);


⸻
