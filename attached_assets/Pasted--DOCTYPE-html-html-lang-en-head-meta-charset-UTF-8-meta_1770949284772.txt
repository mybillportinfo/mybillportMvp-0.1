<script>
const paymentUrls = {
  "Rogers": "https://www.rogers.com/pay-bill",
  "Bell": "https://www.bell.ca/pay",
  "Telus": "https://www.telus.com/pay",
  "Fido": "https://www.fido.ca/payment",
  "Toronto Hydro": "https://www.torontohydro.com/pay",
  "Enbridge": "https://www.enbridgegas.com/pay-bill",
  "Hydro One": "https://www.hydroone.com/pay",
  "Hydro Ottawa": "https://hydroottawa.com/pay",
  "Alectra": "https://www.alectrautilities.com/pay",
  "City of Toronto": "https://www.toronto.ca/property-tax",
  "Canada Revenue Agency": "https://www.canada.ca/pay-cra",
  "Manitoba Hydro": "https://www.hydro.mb.ca/pay",
  "BC Hydro": "https://www.bchydro.com/pay",
  "EPCOR": "https://www.epcor.com/pay",
  "SaskPower": "https://www.saskpower.com/pay",
  "New Brunswick Power": "https://www.nbpower.com/pay",
  "Newfoundland Power": "https://www.newfoundlandpower.com/pay",
  "Maritime Electric": "https://www.maritimeelectric.com/pay"
};

const billerDisplay = document.getElementById("billerDisplay");
const payButton = document.getElementById("payButton");
const searchButton = document.getElementById("searchButton");
const billerSelect = document.getElementById("billerSelect");
const billerInput = document.getElementById("billerInput");

Object.keys(paymentUrls).forEach(biller => {
  const option = document.createElement("option");
  option.value = biller;
  option.textContent = biller;
  billerSelect.appendChild(option);
});

function updateBiller(biller) {
  if (!biller) return;

  billerDisplay.textContent = biller;

  if (paymentUrls[biller]) {
    payButton.style.display = "block";
    searchButton.style.display = "none";
    payButton.textContent = `Pay Now on ${biller} Website`;
    payButton.onclick = () => {
      window.open(paymentUrls[biller], "_blank");
    };
  } else {
    payButton.style.display = "none";
    searchButton.style.display = "block";
    searchButton.onclick = () => {
      const query = encodeURIComponent(`${biller} pay bill Canada`);
      window.open(`https://www.google.com/search?q=${query}`, "_blank");
    };
  }
}

billerSelect.addEventListener("change", e => {
  updateBiller(e.target.value);
});

billerInput.addEventListener("blur", e => {
  updateBiller(e.target.value.trim());
});

const params = new URLSearchParams(window.location.search);
const billerParam = params.get("biller");
if (billerParam) {
  updateBiller(billerParam);
}
</script>
