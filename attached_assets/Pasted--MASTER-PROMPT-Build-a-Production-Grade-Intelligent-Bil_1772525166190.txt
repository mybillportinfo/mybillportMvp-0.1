📦 Layer 1: Deterministic Extraction Engine

1.1 Fuzzy Regex for OCR Resilience

Standard regex fails with OCR imperfections. Implement fuzzy regex using Levenshtein distance to match patterns even with character errors .

```javascript
// fuzzy-regex.js
const levenshtein = require('fast-levenshtein');

function fuzzyMatch(pattern, text, threshold = 0.85) {
  // Convert pattern to regex with wildcards for fuzzy matching
  // Or use library like `fuzzball` for fuzzy string matching
  const distance = levenshtein.get(pattern, text);
  const maxLength = Math.max(pattern.length, text.length);
  const similarity = 1 - (distance / maxLength);
  return {
    match: similarity >= threshold,
    similarity,
    confidence: similarity
  };
}
```

✅ Layer 4: Validation & Enrichment

4.1 Business Rule Validation

```javascript
function validateExtraction(bill) {
  const errors = [];
  
  // Account number format validation
  if (bill.accountNumber?.value && !isValidAccountFormat(bill.accountNumber.value)) {
    errors.push({ field: 'accountNumber', reason: 'Invalid format' });
    bill.accountNumber.confidence *= 0.7;
  }
  
  // Amount reasonability
  if (bill.amountDue?.value && (bill.amountDue.value < 1 || bill.amountDue.value > 10000)) {
    errors.push({ field: 'amountDue', reason: 'Amount outside expected range' });
    bill.amountDue.confidence *= 0.5;
  }
  
  // Due date cannot be in past (unless email is old)
  if (bill.dueDate?.value && new Date(bill.dueDate.value) < new Date()) {
    // Could be valid if email is from previous month
    // Reduce confidence but don't eliminate
    bill.dueDate.confidence *= 0.8;
  }
  
  return { bill, errors };
}
```

📊 Layer 5: Confidence Scoring & Routing

5.1 Multi-Level Confidence Calculation 

```javascript
function calculateOverallConfidence(extraction) {
  const weights = {
    amountDue: 0.4,
    dueDate: 0.3,
    accountNumber: 0.2,
    billerName: 0.1
  };
  
  let weightedSum = 0;
  let totalWeight = 0;
  
  for (const [field, weight] of Object.entries(weights)) {
    if (extraction[field]?.confidence) {
      weightedSum += extraction[field].confidence * weight;
      totalWeight += weight;
    }
  }
  
  // Penalize for missing critical fields
  const missingPenalty = totalWeight < 0.7 ? 0.8 : 1.0;
  
  return (weightedSum / totalWeight) * missingPenalty;
}
```

5.2 Confidence Calibration 

```javascript
// Calibration curve: map raw confidence to actual accuracy
// Based on validation data
const calibrationCurve = {
  0.9: { actualAccuracy: 0.82, action: 'review' },
  0.95: { actualAccuracy: 0.91, action: 'auto_accept' },
  // etc.
};

function calibrateConfidence(rawConfidence, fieldType) {
  // Apply calibration based on field type and historical accuracy
  const calibrated = calibrationCurve[fieldType]?.[Math.round(rawConfidence * 10) / 10] 
    || rawConfidence * 0.9; // Default discount
  
  return Math.min(1, Math.max(0, calibrated));
}
```

```javascript
function routeExtraction(extraction) {
  const confidence = extraction.overallConfidence;
```

🧩 Integration: Complete Scan Endpoint

```javascript
// pages/api/gmail/scan.js
import { getGmailClient } from '../../../lib/gmailClient';
import { runDeterministicExtraction } from '../../../lib/extractors';
import { runClaudeExtraction } from '../../../lib/claudeExtraction';
import { reconcileExtractions } from '../../../lib/reconciliation';
import { validateExtraction, enrichWithBillerInfo } from '../../../lib/validation';
import { calculateConfidence, routeExtraction } from '../../../lib/confidence';
import { storePendingBill } from '../../../lib/firestore';

export default async function handler(req, res) {
  const { userId } = await verifyAuth(req);
  
  try {
    // Fetch emails
    const gmail = await getGmailClient(userId);
    const emails = await fetchBillEmails(gmail, userId);
    
    const results = [];
    
    for (const email of emails) {
      // LAYER 1: Deterministic
      const deterministic = await runDeterministicExtraction(email);
      
      // LAYER 2: AI Fallback (if needed)
      let aiResult = null;
      if (deterministic.overallConfidence < 0.7) {
        aiResult = await runClaudeExtraction(email, deterministic);
      }
      
      // LAYER 3: Reconciliation
      const reconciled = await reconcileExtractions([deterministic, aiResult].filter(Boolean));
      
      // LAYER 4: Validation & Enrichment
      const validated = await validateExtraction(reconciled);
      const enriched = await enrichWithBillerInfo(validated);
      
      // LAYER 5: Final Confidence & Routing
      enriched.overallConfidence = calculateOverallConfidence(enriched);
      const routing = routeExtraction(enriched);
      
      // Store for user review
      const pendingId = await storePendingBill(userId, enriched, routing);
      
      results.push({
        emailId: email.id,
        extraction: enriched,
        routing,
        pendingId
      });
    }
    
    return res.status(200).json({
      success: true,
      processed: results.length,
      pendingCount: results.filter(r => r.routing.action !== 'reject').length,
      results
    });
    
  } catch (error) {
    console.error('Scan failed:', error);
    return res.status(500).json({
      success: false,
      error: 'Scan failed',
      message: error.message
    });
  }
}
```
